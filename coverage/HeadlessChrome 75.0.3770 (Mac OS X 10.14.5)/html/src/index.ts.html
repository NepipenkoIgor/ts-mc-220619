    return typeof item === 'string';

export function average(...args: sn[]): string {
    let total: number = 0;
    for (const item of args) {
        if (isString(item)) {
            total += Number(item);
            continue;
        }
        total += item;
    }
    const avg: number = total / args.length;
    return `AVERAGE IS ${avg}`;
}
